<block>::= <operation> <block> | <operation>

<operation>::= <attribute> | <declaration> | <if> | <while> | <jumppoint> | <goto>

<operand>::= ID | number | ' letter '
<mul-exp>::= <expression> * <expression>
<add-exp>::= <expression> + <expression>
<sub-exp>::= <expression> - <expression>
<div-exp>::= <expression> / <expression>
<not-exp>::= ! <expression>
<eq-exp>::= <expression> == <expression>
<neq-exp>::= <expression != <expression>
<leq-exp>::= <expression <= <expression>
<geq-exp>::= <expression >= <expression>
<lt-exp>::= <expression < <expression>
<gt-exp>::= <expression > <expression>
<expression>::= <operand> | <mul-exp> | <add-exp> | <sub-exp> | <div-exp> | <not-exp> | <eq-exp> | <neq-exp> | <leq-exp> | <geq-exp> | <lt-exp> | <gt-exp> | (<expression>)

<assignment>::= = <expression> ;
<attribute>::= ID <assignment> ;

<type>::= int | char
<declaration>::= <type> ID | <type> ID <assignment>;

<while>::= while ( <expression> ) { <block> }
<if>::= if ( <expression> ) { <block> } else <else> | if ( <expression> ) { <block> }
<else>::= <if> | { <block> }

<jumppoint>::= ID :
<goto>::= goto ID ;
